编译mDolphin（基于Ubuntu12.04）:
从github下载代码（https://github.com/VincentWei/mdolphin-core）；
然后：
sudo apt-get install libxml2-dev libcurl4-openssl-dev libssl-dev libxslt-dev libc-ares-dev
sudo apt-get install gtk-doc-tools 
sudo apt-get install libjpeg-dev

unzip mdolphin-core-master.zip 
cd mdolphin-core-master/ 
./autogen.sh 
./configure CFLAGS=-lstdc++ CPPFLAGS=-fpermissive --enable-canvas

防止后面编译出错，执行：
cd /usr/include
sudo ln -s libxml2/libxml/ libxml

修改GNUMakefile中的libJavaScriptCore_la_LINK，添加-licuuc，如下：
libJavaScriptCore_la_LINK = $(LIBTOOL) $(AM_V_lt) --tag=CXX \
        $(AM_LIBTOOLFLAGS) $(LIBTOOLFLAGS) --mode=link $(CXXLD) \
        $(libJavaScriptCore_la_CXXFLAGS) $(CXXFLAGS) $(AM_LDFLAGS) \
        -licuuc \
        $(LDFLAGS) -o $@

然后编译：
make

void BackendSaveDataFile()
{
	if(!Check_FileSaveData())
		return;

	char path[200];
	char file_name[100];

	snprintf(file_name, sizeof(file_name), "%s%s"
			, GetCfgValue(save_file_name_cfg).c_str(), GetCfgValue(save_file_index_cfg).c_str());
	SetCfgValue(SaveNameCfg, file_name);

	FILE_DATA_TYPE type = CheckDataType();
	const char * type_str = GetFileTypeExtStr(GetFileTypeNum(type));

	int i = 0;
	char append_str[100] = "";
	do
	{
		snprintf(path, sizeof(path), "%s%s%s%s%s"
				, GetUserSaveDataPath()
				, GetCfgValue(SaveNameCfg).c_str()
				, append_str
				, type_str, CSV_EXT);

		snprintf(append_str, sizeof(append_str), "-%d", ++i);
	}
	while(access(path, F_OK) == 0);

	SaveDataFile(REPORT_DEVISER, path);

	UpdateFileName();
}
